# HOW TO INSTALL
#   python -m venv .venv
#   source .venv/bin/activate          # Windows: .venv\Scripts\activate
#   pip install -r requirements.txt
#
# HOW THIS FILE IS ORGANISED
#   Each package has an inline comment explaining:
#     • WHERE it is used (which module / file)
#     • WHY it is needed (what it does for this project)
#   Packages are grouped by layer so you can quickly find what you need.

# PINNING POLICY
#   All versions are pinned with == (exact) rather than >= (minimum) so that

# ────────────────────────────────────────────────────────────────────────────
# 1. CORE AGENT FRAMEWORK
#    The multi-agent orchestration layer that runs Coordinator → Researcher
#    → Analyst → Fact-Checker → Writer in a sequential CrewAI pipeline.
# ────────────────────────────────────────────────────────────────────────────

crewai==1.15.2
# The main CrewAI package.  Provides Agent, Task, Crew, Process, and LLM
# classes used throughout backend/crew.py.  Version 1.15.2 is the earliest
# release that supports the per-run crew factory pattern (avoiding the
# "Executor is already running" error on back-to-back requests).

crewai-core==1.15.2
# Low-level internals shared by crewai and crewai-tools (event bus, base
# agent executor, token process counter).  Must match crewai exactly.

crewai-tools==1.15.2
# Provides crewai.tools.BaseTool — the base class for SafeSearchTool in
# backend/tools/safe_search_tool.py.  Must match crewai exactly.

crewai-cli==1.15.2
# CLI helper shipped alongside crewai (crewai create, crewai run, etc.).
# Installed as a dependency of crewai; not directly invoked by this project
# but must be version-locked to avoid resolver conflicts.


# ────────────────────────────────────────────────────────────────────────────
# 2. LLM PROVIDERS & INFERENCE
#    Everything needed to call Groq (primary), OpenAI-compatible endpoints,
#    and Google Gemini through a unified litellm interface.
# ────────────────────────────────────────────────────────────────────────────

litellm==1.92.0
# Universal LLM proxy used by CrewAI internally.  All model calls in
# backend/crew.py go through litellm.completion().  Provides:
#   • Unified API across Groq / OpenAI / Anthropic / Gemini
#   • num_retries and request_timeout config (set in crew.py to handle
#     Groq free-tier 429 rate-limit errors automatically)
#   • Token counting and context-window enforcement
# Pinned to 1.92.0 — earlier versions do not honour Groq's Retry-After
# header correctly, causing premature retries and cascading 429s.

groq==1.5.0
# Official Groq Python SDK.  litellm delegates to this for
# groq/llama-3.3-70b-versatile calls.  Also used directly in backend/crew.py
# to monkey-patch the mark_cache_breakpoint no-op (Groq rejects the
# 'cache_breakpoint' extra field that CrewAI injects for Anthropic).

openai==2.45.0
# OpenAI Python SDK — required by litellm even when OpenAI is not the
# primary provider, because litellm's internal message formatting and
# function-call parsing are based on the OpenAI message schema.

google-genai==1.65.0
# Google Generative AI SDK — pulled in by crewai[google-genai] for optional
# Gemini model support.  Not used in the default Groq configuration but
# must be installed to satisfy crewai's import graph.

instructor==1.15.4
# Structured output extraction library used by crewai-tools internally when
# parsing agent responses into typed Pydantic objects.  Not called directly
# in this project's code but required by the crewai dependency tree.

tiktoken==0.12.0
# OpenAI's fast BPE tokeniser.  litellm uses it to count prompt tokens
# before sending requests (to enforce max_tokens limits and detect context
# overflow before hitting the API).

tokenizers==0.23.1
# HuggingFace tokenizers library — used by litellm and crewai-tools for
# models whose token counts are estimated with HF vocabularies rather than
# tiktoken (e.g. Llama-family models on Groq).
